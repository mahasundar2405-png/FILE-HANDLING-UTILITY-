import java.io.*; public class FileHandlingUtility { // Method to write data into a file public static void writeFile(String filename) { try { FileWriter fw = new FileWriter(filename); fw.write("CODTECH Internship Program\n"); fw.write("File Handling Utility using Java\n"); fw.write("This file demonstrates write operation.\n"); fw.close(); System.out.println("File written successfully.\n"); } catch (IOException e) { System.out.println("Error while writing file."); } } // Method to read data from a file public static void readFile(String filename) { try { FileReader fr = new FileReader(filename); BufferedReader br = new BufferedReader(fr); String line; System.out.println("Reading file contents:"); while ((line = br.readLine()) != null) { System.out.println(line); } br.close(); System.out.println(); } catch (IOException e) { System.out.println("Error while reading file."); } } // Method to modify (append) data into a file public static void modifyFile(String filename) { try { FileWriter fw = new FileWriter(filename, true); // true = append mode BufferedWriter bw = new BufferedWriter(fw); bw.write("\nThis line is added using modify operation."); bw.close(); System.out.println("File modified successfully.\n"); } catch (IOException e) { System.out.println("Error while modifying file."); } } // Main method public static void main(String[] args) { String filename = "codtech_file.txt"; writeFile(filename); // Writing to file readFile(filename); // Reading file modifyFile(filename); // Modifying file readFile(filename); // Reading modified file } }
